Cascader 级联选择

使用指南

  • 确保 components 下引入 transition、icon、cascader组件
  • 在 app.json 或 index.json 中引入组件
{
  "usingComponents": {
    "fs-cascader": "path/components/cascader/index"
  }
}

<fs-cascader
  currentList="{{addressList}}"

  show="{{show}}"
  level="{{level}}"
  bind:close="toggleCascader"
  bind:change="handleChange">
</fs-cascader>

const app = getApp()

Page({
  data: {
    addressString: '',
    address: [], // 已选地址列表
    addressList: [], // 当前可选地址列表
    show: false, // 是否显示
    level: 3 // 级数
  },
  onLoad () {
    this.requestAddress([])
  },
  toggleCascader (e) {
    this.setData({
      show: !this.data.show
    })
  },
  handleChange (e) {
    this.setData({
      address: e.detail,
      addressString: JSON.stringify(e.detail)
    })
    // 更新地址
    this.requestAddress(e.detail)
  },
  /**
   * 获取地址
   * @param {Array} val [{label: '', value: ''}]
   */
  requestAddress (val) {
    const data = [
      {
        label: '城区',
        value: '1000',
        child: [
          {
            label: '城区1',
            value: '1100',
            child: [
              {
                label: '城区1_1',
                value: '1101'
              }, {
                label: '城区1_2',
                value: '1102'
              }
            ]
          }, {
            label: '城区2',
            value: '1200',
            child: [
              {
                label: '城区2_1',
                value: '1201'
              }, {
                label: '城区2_2',
                value: '1202'
              }
            ]
          }
        ]
      }, {
        label: '郊区',
        value: '2000',
        child: [
          {
            label: '郊区1',
            value: '2100',
            child: [
              {
                label: '郊区1_1',
                value: '2101'
              }, {
                label: '郊区1_2',
                value: '2102'
              }
            ]
          }, {
            label: '郊区2',
            value: '2200',
            child: [
              {
                label: '郊区2_1',
                value: '2201'
              }, {
                label: '郊区2_2',
                value: '2202'
              }
            ]
          }
        ]
      }
    ]
    let addressList = []
    if (val.length === 0) {
      addressList = data.map((item) => ({
        label: item.label,
        value: item.value
      }))
    } else if (val.length === 1) {
      for (let i = 0; i < data.length; i++) {
        if (val[0].value === data[i].value) {
          addressList = data[i].child.map((item) => {
            return {
              label: item.label,
              value: item.value
            }
          })
        }
      }
    } else if (val.length === 2) {
      for (let i = 0; i < data.length; i++) {
        if (val[0].value === data[i].value) {
          for (let j = 0; j < data[i].child.length; i++) {
            if (val[1].value === data[i].child[j].value) {
              addressList = data[i].child[j].child
              break
            }
          }
        }
      }
    }
    setTimeout(() => {
      this.setData({
        addressList
      })
    }, 500)
  }
})

参数

参数 说明 类型 默认值

selectedList 当前选择的地址数据 Array []
show 是否显示 Boolean false

事件

事件 说明
change 数值变化时调用，在这个事件中获取数据源
close 关闭时事件

获取源数据示例

requestAddress ({ len, id }) {
  axios.post('/get-city-json', `type=${len}&code=${id}`).then(res => {
    this.currentData = res.data
  })
}
